1.Find the Smaller Angle

        var Minimal_Angle = (h, m) => 
        {
            h = h % 12  ;
            hh = (h * 360) /12 + (m * 360)/(12 * 60);
            mm = (m * 360) / (60);
            var angle = Math.abs(hh - mm);
            if (angle > 180){
              angle = 360 - angle;
              }
              return angle;
        };

3.Perfect Number Check.

var Perfect_Check = (N) => 
        {
          let sum = 0;
          for(let i = 1; i <= N/2 ; i++){
            if(N%i === 0){
              sum += i;
            }
          }
          if(sum === N && sum !== 0){
            return "YES";
          } 
          else{
            return "NO";
          }
        };

        var Reverse_Number = (N) => 
        {
          let rev = parseFloat(N.toString().split('').reverse().join(''))
          return rev;
        
        };
